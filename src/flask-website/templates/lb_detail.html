Fichier de configuration Nginx 

upstream backend_{{ lb.id }} {
    server {{ lb.pass | replace('http://', '') | replace('https://', '') }};
}

server {
    listen {{ lb.ip_bind }}:80;
    server_name {{ lb.name | lower | replace(' ', '_') }}.local;

    location / {
        proxy_pass http://backend_{{ lb.id }};
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
🚀 Préparation au déploiement
Méthode 1 : Conteneurisation (Docker)
# 1. Télécharger le fichier .conf dans le dossier courant
# 2. Lancer le conteneur Nginx

docker run -d --name nginx-lb-{{ lb.id }} \
  -p 80:80 \
  -v $(pwd)/nginx_{{ lb.name }}.conf:/etc/nginx/conf.d/default.conf:ro \
  nginx:alpine
Méthode 2 : Installation native (Debian / AlmaLinux)
# 1. Installer Nginx
sudo apt update && sudo apt install nginx -y # Debian/Ubuntu
# sudo dnf install nginx -y # AlmaLinux

# 2. Déployer la configuration
sudo cp nginx_{{ lb.name }}.conf /etc/nginx/conf.d/
sudo nginx -t && sudo systemctl restart nginx
